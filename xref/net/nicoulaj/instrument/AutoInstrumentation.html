38  /**
39   * Provides transparent access to the {@link Instrumentation} API, without explicitly starting the application with an agent,
40   * attaching one at runtime or even implementing one.
41   * <p/>
42   * <b>Disclaimer:</b> This implementation is a proof of concept using several hacks that are not guaranteed to be consistent
43   * across systems and JVMs. Do not rely on it for production applications !
44   *
45   * @author Julien Nicoulaud <julien.nicoulaud@gmail.com>
46   * @see #getInstrumentation() API user entry point
47   * @since 1.0.0
48   */
49  public class AutoInstrumentation {
50  
51      /**
52       * The instance of {@link Instrumentation} retrieved through the agent.
53       *
54       * @see #getInstrumentation()
55       */
56      private static Instrumentation instrumentation;
57  
58      /**
59       * Get an instance of {@link Instrumentation}.
60       * <p/>
61       * Any error occured while retrieving the instance is thrown as {@link RuntimeException}.
62       *
63       * @return an instance of {@link Instrumentation}
64       */
65      @SuppressWarnings("unused") // Public API
66      synchronized public static Instrumentation getInstrumentation() {
67          if (instrumentation == null) {
68              try {
69  
70                  // Generate agent jar
71                  final File agentJarFile = generateAgentJar();
72  
73                  // Get current process id
74                  final int pid = getPid();
75  
76                  // Load VirtualMachine class
77                  final Class vmClass = getVirtualMachineClass();
78  
79                  // Attach to virtual machine
80                  final Object vm = vmClass.getDeclaredMethod("attach", String.class).invoke(vmClass, String.valueOf(pid));
81  
82                  // Load agent
83                  vmClass.getMethod("loadAgent", String.class).invoke(vm, agentJarFile.getPath());
84  
85                  // Detach
86                  vmClass.getMethod("detach").invoke(vm);
87  
88              } catch (Exception e) {
89                  throw new RuntimeException("Failed loading instrumentation", e);
90              }
91          }
92          return instrumentation;
93      }

100      * @param args agent arguments, unused
101      * @param inst instrument object
102      */
103     @SuppressWarnings("unused") // Agent entry point
104     public static void agentmain(String args, Instrumentation inst) {
105         instrumentation = inst;

108     /**
109      * Generate an agent jar file in {@code java.io.tmpdir} with {@code MANIFEST} registering this class as an agent.
110      * <p/>
111      * This method is not meant to be used directly, see {@link #getInstrumentation()}.
112      *
113      * @return the generated jar file object
114      * @throws Exception if the file could not be created or written to, or this class binary stream could not be read
115      * @see #getInstrumentation()
116      */
117     static File generateAgentJar() throws Exception {
118         JarOutputStream jarOS = null;
119         try {
120             // Prepare generated jar output file
121             final File jarFile = File.createTempFile(AutoInstrumentation.class.getSimpleName(), ".jar");
122 
123             // Prepare manifest
124             final Manifest manifest = new Manifest();
125             manifest.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION, "1.0");
126             manifest.getMainAttributes().put(new Attributes.Name("Agent-Class"), AutoInstrumentation.class.getName());
127             manifest.getMainAttributes().put(new Attributes.Name("Can-Redefine-Classes"), Boolean.TRUE.toString());
128             manifest.getMainAttributes().put(new Attributes.Name("Can-Retransform-Classes"), Boolean.TRUE.toString());
129             manifest.getMainAttributes().put(new Attributes.Name("Can-Set-Native-Method-Prefix"), Boolean.TRUE.toString());
130 
131             // Prepare jar output stream
132             jarOS = new JarOutputStream(new FileOutputStream(jarFile), manifest);
133 
134             // Return generated jar file
135             return jarFile;
136 
137         } catch (Throwable t) {
138             throw new Exception("Failed generating agent jar", t);
139         } finally {
140             if (jarOS != null) jarOS.close();
141         }
142     }

144     /**
145      * Get current process PID.
146      * <p/>
147      * Tries two methods, the most portable first.
148      * <p/>
149      * This method is not meant to be used directly, see {@link #getInstrumentation()}.
150      *
151      * @return the current process PID
152      * @throws Exception if the PID could not be retrieved
153      * @see #getPidFromRuntimeBeanName()
154      * @see #getPidFromVMManagement()
155      */
156     static int getPid() throws Exception {
157         try {
158             try {
159                 return getPidFromRuntimeBeanName();
160             } catch (Throwable t) {
161                 return getPidFromVMManagement();
162             }
163         } catch (Throwable t) {
164             throw new Exception("Failed getting process PID", t);
165         }
166     }

168     /**
169      * Get current process PID from {@link RuntimeMXBean} name.
170      * <p/>
171      * Uses a trick: {@code java.lang.Runtime} mbean is named {@code PID@HOSTNAME}.
172      * <p/>
173      * This method is not meant to be used directly, see {@link #getInstrumentation()}.
174      *
175      * @return the current process PID
176      * @throws Exception if the PID could not be retrieved
177      */
178     static int getPidFromRuntimeBeanName() throws Exception {
179         try {
180             String pid = ManagementFactory.getRuntimeMXBean().getName();
181             if (pid.contains("@")) pid = pid.substring(0, pid.indexOf("@"));
182             return Integer.parseInt(pid);
183         } catch (Throwable t) {
184             throw new Exception("Failed getting process PID", t);
185         }
186     }
187 
188     /**
189      * Get current process PID from {@code VMManagement} instance.
190      * <p/>
191      * Only tested on HotSpot JVM.
192      * <p/>
193      * This method is not meant to be used directly, see {@link #getInstrumentation()}.
194      *
195      * @return the current process PID
196      * @throws Exception if the PID could not be retrieved
197      */
198     static int getPidFromVMManagement() throws Exception {
199         try {
200             final RuntimeMXBean runtimeMXBean = ManagementFactory.getRuntimeMXBean();
201             final Field jvmField = runtimeMXBean.getClass().getDeclaredField("jvm");
202             jvmField.setAccessible(true);
203             Object vmManagement = jvmField.get(runtimeMXBean);
204             final Method getProcessIdMethod = vmManagement.getClass().getDeclaredMethod("getProcessId");
205             getProcessIdMethod.setAccessible(true);
206             return (Integer) getProcessIdMethod.invoke(vmManagement);
207         } catch (Throwable t) {
208             throw new Exception("Failed getting process PID from VManagement", t);
209         }
210     }

212     /**
213      * Load Java's attach API {@code VirtualMachine} class.
214      * <p/>
215      * Tries current class classloader, then tries after adding {@code tools.jar} to classpath.
216      * <p/>
217      * This method is not meant to be used directly, see {@link #getInstrumentation()}.
218      *
219      * @return the {@link Class} object
220      * @throws Exception if the class could not be loaded
221      * @see #getVirtualMachineClassFromClassLoader()
222      * @see #addToolsJarToClassPath()
223      */
224     static Class getVirtualMachineClass() throws Exception {
225         try {
226             try {
227                 return getVirtualMachineClassFromClassLoader();
228             } catch (ClassNotFoundException e) {
229                 addToolsJarToClassPath();
230                 return getVirtualMachineClassFromClassLoader();
231             }
232         } catch (Throwable t) {
233             throw new Exception("Failed loading attach API class", t);
234         }
235     }

239      * <p/>
240      * This method is not meant to be used directly, see {@link #getInstrumentation()}.

243      * @throws ClassNotFoundException if the class could not be found in classloader

247         return AutoInstrumentation.class.getClassLoader().loadClass("com.sun.tools.attach.VirtualMachine");

250     /**
251      * Locate and add {@code tools.jar} (or {@code classes.jar} on Mac OS) to classpath.
252      * <p/>
253      * This method is not meant to be used directly, see {@link #getInstrumentation()}.
254      *
255      * @throws Exception if {@code tools.jar} could not be located or added to classpath
256      * @see #getVirtualMachineClass()
257      */
258     static void addToolsJarToClassPath() throws Exception {
259         final boolean isMac = System.getProperty("os.name").toLowerCase().contains("mac");
260         try {
261             // Locate tools.jar
262             URL url = null;
263             MalformedURLException mfe = null;
264             try {
265                 url = new URL(new File(System.getProperty("java.home")).toURI().toString() + "/../" + (isMac ? "Classes/classes.jar" : "lib/tools.jar"));
266             } catch (MalformedURLException e) {
267                 mfe = e;
268             }
269             if (mfe != null || url == null || !new File(url.toURI()).isFile())
270                 throw new Exception("Failed locating " + (isMac ? "classes.jar" : "tools.jar") + ", make sure you run this program on a JDK or explicitly add it to classpath", mfe);
271 
272             // Add it to classpath
273             final Method addURL = URLClassLoader.class.getDeclaredMethod("addURL", URL.class);
274             addURL.setAccessible(true);
275             addURL.invoke(AutoInstrumentation.class.getClassLoader(), url);
276 
277         } catch (Throwable t) {
278             throw new Exception("Failed adding " + (isMac ? "classes.jar" : "tools.jar") + " to classpath", t);
279         }
280     }
